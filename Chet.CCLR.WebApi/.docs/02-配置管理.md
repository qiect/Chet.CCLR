# 配置管理

## 1. 配置概述

本文档详细描述了Chet.CCLR.WebApi项目的配置管理系统，包括配置源、配置结构、环境管理以及配置的最佳实践。本项目采用.NET 8内置的Configuration系统，支持多种配置源和灵活的环境管理。

## 2. 配置源层级

.NET Configuration系统按照以下优先级顺序加载配置：

1. **appsettings.Production.json** (最高优先级)
2. **appsettings.Staging.json**
3. **appsettings.Development.json**
4. **appsettings.json** (默认配置)
5. **环境变量**
6. **命令行参数**

## 4. 配置模型

### 4.1 JWT配置模型

```csharp
public class JwtSettings
{
    public string Secret { get; set; } = string.Empty;
    public string Issuer { get; set; } = string.Empty;
    public string Audience { get; set; } = string.Empty;
    public int TokenExpirationInMinutes { get; set; } = 60;
    
    public bool IsValid()
    {
        return !string.IsNullOrWhiteSpace(Secret) &&
               !string.IsNullOrWhiteSpace(Issuer) &&
               !string.IsNullOrWhiteSpace(Audience) &&
               TokenExpirationInMinutes > 0;
    }
}
```

### 4.2 缓存配置模型

```csharp
public class CacheSettings
{
    public string RedisConnectionString { get; set; } = "localhost:6379";
    public int DefaultAbsoluteExpirationMinutes { get; set; } = 60;
    public int SlidingExpirationMinutes { get; set; } = 30;
    public bool EnableDistributedCache { get; set; } = true;
    
    public TimeSpan GetDefaultExpiry() => TimeSpan.FromMinutes(DefaultAbsoluteExpirationMinutes);
    public TimeSpan GetSlidingExpiry() => TimeSpan.FromMinutes(SlidingExpirationMinutes);
}
```

### 4.3 应用程序设置模型

```csharp
public class AppSettings
{
    public string AppName { get; set; } = "Chet.CCLR.WebApi";
```

## 5. 配置加载与验证

### 5.1 Program.cs 配置加载

```csharp
var builder = WebApplication.CreateBuilder(args);

// 加载配置
ConfigureConfiguration(builder);
ConfigureServices(builder);
ConfigureMiddlewares(builder);

var app = builder.Build();

ConfigurePipeline(app);

app.Run();

static void ConfigureConfiguration(WebApplicationBuilder builder)
{
    // 配置日志
    builder.Host.UseSerilog((context, configuration) =>
        context.ConfigureSerilog(configuration)
    );

    // 添加自定义配置源（如Azure Key Vault, Consul等）
    if (builder.Environment.IsProduction())
    {
        // 在生产环境中可以从外部配置源加载敏感配置
        // builder.Configuration.AddAzureKeyVault(...);
    }
}

static void ConfigureServices(WebApplicationBuilder builder)
{
    // 配置验证
    var jwtSettings = builder.Configuration.GetSection("JwtSettings").Get<JwtSettings>();
    if (jwtSettings == null || !jwtSettings.IsValid())
    {
        throw new InvalidOperationException("JWT配置无效");
    }

    // 注册配置选项
    builder.Services.Configure<JwtSettings>(builder.Configuration.GetSection("JwtSettings"));
    builder.Services.Configure<CacheSettings>(builder.Configuration.GetSection("CacheSettings"));
    builder.Services.Configure<AppSettings>(builder.Configuration.GetSection("AppSettings"));

    // 注册服务
    builder.Services.AddScoped<IUserService, UserService>();
    builder.Services.AddScoped<IUserRepository, UserRepository>();
    
    // 注册AutoMapper
    builder.Services.AddAutoMapper(typeof(Program));
    
    // 注册数据库上下文
    builder.Services.AddDbContext<AppDbContext>(options =>
    {
        if (builder.Environment.IsDevelopment())
        {
            options.UseSqlite(builder.Configuration.GetConnectionString("DefaultConnection"));
        }
        else
        {
            options.UseNpgsql(builder.Configuration.GetConnectionString("PostgreSqlConnection"));
        }
    });
    
    // 注册JWT认证
    builder.Services.AddAuthentication(options =>
    {
        options.DefaultAuthenticateScheme = JwtBearerDefaults.AuthenticationScheme;
        options.DefaultChallengeScheme = JwtBearerDefaults.AuthenticationScheme;
    })
    .AddJwtBearer(options =>
    {
        var jwtSettings = builder.Configuration.GetSection("JwtSettings").Get<JwtSettings>();
        
        options.TokenValidationParameters = new TokenValidationParameters
        {
            ValidateIssuer = true,
            ValidateAudience = true,
            ValidateLifetime = true,
            ValidateIssuerSigningKey = true,
            ValidIssuer = jwtSettings.Issuer,
            ValidAudience = jwtSettings.Audience,
            IssuerSigningKey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(jwtSettings.Secret)),
            ClockSkew = TimeSpan.Zero
        };
    });
}

static void ConfigureMiddlewares(WebApplicationBuilder builder)
{
    if (builder.Environment.IsDevelopment())
    {
        builder.Services.AddSwaggerGen(c =>
        {
            c.SwaggerDoc("v1", new OpenApiInfo { Title = "Chet.CCLR.WebApi", Version = "v1" });
            
            // 添加JWT认证支持
            c.AddSecurityDefinition("Bearer", new OpenApiSecurityScheme
            {
                Description = "JWT Authorization header using the Bearer scheme",
                Name = "Authorization",
                In = ParameterLocation.Header,
                Type = SecuritySchemeType.ApiKey,
                Scheme = "Bearer"
            });
            
            c.AddSecurityRequirement(new OpenApiSecurityRequirement
            {
                {
                    new OpenApiSecurityScheme
                    {
                        Reference = new OpenApiReference
                        {
                            Type = ReferenceType.SecurityScheme,
                            Id = "Bearer"
                        }
                    },
                    new string[] {}
                }
            });
        });
    }

    builder.Services.AddControllers();
    builder.Services.AddEndpointsApiExplorer();
}

static void ConfigurePipeline(WebApplication app)
{
    if (app.Environment.IsDevelopment())
    {
        app.UseSwagger();
        app.UseSwaggerUI();
    }

    app.UseHttpsRedirection();
    app.UseAuthentication();
    app.UseAuthorization();
    app.MapControllers();
}
```

### 5.2 配置扩展方法

```csharp
public static class ConfigurationExtensions
{
    public static void ConfigureSerilog(this HostBuilderContext context, LoggerConfiguration configuration)
    {
        configuration.ReadFrom.Configuration(context.Configuration);
    }

    public static T GetRequiredSection<T>(this IConfiguration configuration, string key) where T : class, new()
    {
        var section = configuration.GetSection(key);
        if (section == null)
        {
            throw new InvalidOperationException($"Configuration section '{key}' not found.");
        }

        var options = section.Get<T>();
        if (options == null)
        {
            throw new InvalidOperationException($"Configuration section '{key}' could not be mapped to type {typeof(T).Name}.");
        }

        return options;
    }

    public static bool TryGetSection<T>(this IConfiguration configuration, string key, out T? options) where T : class, new()
    {
        var section = configuration.GetSection(key);
        options = section?.Get<T>();
        return options != null;
    }
}
```

## 7. 配置验证

### 7.1 配置验证中间件

```csharp
public class ConfigurationValidationMiddleware
{
    private readonly RequestDelegate _next;
    private readonly IConfiguration _configuration;
    private readonly ILogger<ConfigurationValidationMiddleware> _logger;

    public ConfigurationValidationMiddleware(
        RequestDelegate next, 
        IConfiguration configuration,
        ILogger<ConfigurationValidationMiddleware> logger)
    {
        _next = next;
        _configuration = configuration;
        _logger = logger;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        // 验证关键配置是否存在
        ValidateCriticalConfigurations();

        await _next(context);
    }

    private void ValidateCriticalConfigurations()
    {
        // 验证JWT配置
        var jwtSettings = _configuration.GetSection("JwtSettings").Get<JwtSettings>();
        if (jwtSettings == null || !jwtSettings.IsValid())
        {
            _logger.LogCritical("JWT配置无效，请检查JwtSettings配置节");
        }

        // 验证数据库连接字符串
        var connectionString = _configuration.GetConnectionString("DefaultConnection");
        if (string.IsNullOrEmpty(connectionString))
        {
            _logger.LogCritical("数据库连接字符串未配置");
        }

        // 验证其他关键配置...
    }
}

// 在Startup中注册中间件
app.UseMiddleware<ConfigurationValidationMiddleware>();
```

### 7.2 启动时配置验证

```csharp
public static class ConfigurationValidator
{
    public static void ValidateConfiguration(WebApplicationBuilder builder)
    {
        var configuration = builder.Configuration;
        
        // 验证JWT设置
        var jwtSettings = configuration.GetSection("JwtSettings").Get<JwtSettings>();
        ValidateJwtSettings(jwtSettings);

        // 验证数据库连接
        ValidateDatabaseConnections(configuration);

        // 验证缓存设置
        ValidateCacheSettings(configuration);

        // 验证其他设置...
    }

    private static void ValidateJwtSettings(JwtSettings? settings)
    {
        if (settings == null)
        {
            throw new InvalidOperationException("JWT设置未配置");
        }

        if (string.IsNullOrWhiteSpace(settings.Secret))
        {
            throw new InvalidOperationException("JWT Secret未配置");
        }

        if (settings.Secret.Length < 32)
        {
            throw new InvalidOperationException("JWT Secret长度至少需要32个字符");
        }

        if (string.IsNullOrWhiteSpace(settings.Issuer))
        {
            throw new InvalidOperationException("JWT Issuer未配置");
        }

        if (string.IsNullOrWhiteSpace(settings.Audience))
        {
            throw new InvalidOperationException("JWT Audience未配置");
        }

        if (settings.TokenExpirationInMinutes <= 0)
        {
            throw new InvalidOperationException("JWT TokenExpirationInMinutes必须大于0");
        }
    }

    private static void ValidateDatabaseConnections(IConfiguration configuration)
    {
        var defaultConnection = configuration.GetConnectionString("DefaultConnection");
        var postgresqlConnection = configuration.GetConnectionString("PostgreSqlConnection");

        if (string.IsNullOrEmpty(defaultConnection) && string.IsNullOrEmpty(postgresqlConnection))
        {
            throw new InvalidOperationException("未配置数据库连接字符串");
        }
    }

    private static void ValidateCacheSettings(IConfiguration configuration)
    {
        var cacheSettings = configuration.GetSection("CacheSettings").Get<CacheSettings>();
        if (cacheSettings == null)
        {
            throw new InvalidOperationException("缓存设置未配置");
        }

        if (cacheSettings.EnableDistributedCache && string.IsNullOrEmpty(cacheSettings.RedisConnectionString))
        {
            throw new InvalidOperationException("启用分布式缓存但Redis连接字符串未配置");
        }
    }
}
```

## 8. 配置管理最佳实践

### 8.1 配置分离

- **敏感配置**: 使用环境变量或外部配置源（如Azure Key Vault）
- **环境特定配置**: 使用环境特定的配置文件
- **共享配置**: 在appsettings.json中定义默认值

### 8.2 配置加密

对于敏感配置，可以使用.NET的配置加密功能：

```csharp
// 在部署脚本中加密配置
dotnet user-secrets set "JwtSettings:Secret" "your-secret-value"
```

### 8.3 配置热重载

```csharp
// 支持配置热重载
builder.Configuration.AddJsonFile("appsettings.json", optional: false, reloadOnChange: true);
```

```csharp
public class ConfigurationAuditService
{
    private readonly ILogger<ConfigurationAuditService> _logger;
    private readonly IConfiguration _configuration;

    public ConfigurationAuditService(ILogger<ConfigurationAuditService> logger, IConfiguration configuration)
    {
        _logger = logger;
        _configuration = configuration;
    }

    public void LogConfigurationState()
    {
        var appSettings = _configuration.GetSection("AppSettings").Get<AppSettings>();
        var jwtSettings = _configuration.GetSection("JwtSettings").Get<JwtSettings>();

            jwtSettings?.TokenExpirationInMinutes, jwtSettings?.Issuer);
    }
}
```

```csharp
public class ConfigurationTests
{
    [Fact]
    public void JwtSettings_ConfigurationMapping_Success()
    {
        // Arrange
        var config = new ConfigurationBuilder()
            .AddInMemoryCollection(new Dictionary<string, string?>
            {
                {"JwtSettings:Secret", "very-long-secret-key-for-testing"},
                {"JwtSettings:Issuer", "test-issuer"},
                {"JwtSettings:Audience", "test-audience"},
                {"JwtSettings:TokenExpirationInMinutes", "60"}

            .Build();

        // Act
        var jwtSettings = config.GetSection("JwtSettings").Get<JwtSettings>();

        // Assert
        jwtSettings.Should().NotBeNull();
        jwtSettings!.Secret.Should().Be("very-long-secret-key-for-testing");
        jwtSettings.Issuer.Should().Be("test-issuer");
        jwtSettings.Audience.Should().Be("test-audience");
        jwtSettings.TokenExpirationInMinutes.Should().Be(60);
        jwtSettings.IsValid().Should().BeTrue();
    }

    [Fact]
    public void JwtSettings_InvalidConfiguration_FailsValidation()
    {
        // Arrange
        var config = new ConfigurationBuilder()
            .AddInMemoryCollection(new Dictionary<string, string?>
            {
                {"JwtSettings:Secret", ""},
                {"JwtSettings:Issuer", ""},
                {"JwtSettings:Audience", ""},
                {"JwtSettings:TokenExpirationInMinutes", "0"}
            })
            .Build();

        // Act
        var jwtSettings = config.GetSection("JwtSettings").Get<JwtSettings>();

    }
```
