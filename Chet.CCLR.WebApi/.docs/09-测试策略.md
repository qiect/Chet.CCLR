# 测试策略

## 1. 测试概述

本文档详细描述了Chet.CCLR.WebApi项目的测试策略，包括测试类型、测试框架、测试结构和最佳实践。本项目采用分层测试策略，涵盖单元测试、集成测试和端到端测试。

## 2. 测试框架和技术栈

### 2.1 单元测试
- **测试框架**: xUnit.net
- **模拟框架**: Moq
- **断言库**: FluentAssertions
- **测试运行器**: .NET Test Explorer

### 2.2 集成测试
- **测试框架**: xUnit.net
- **Web测试基类**: WebApplicationFactory<T>
- **数据库测试**: TestContainers (用于真实数据库测试)

### 2.3 端到端测试
- **测试框架**: Playwright
- **浏览器自动化**: Chromium, Firefox, WebKit

## 4. 单元测试

### 4.1 测试原则

- **独立性**: 每个测试都应该独立运行，不依赖其他测试
- **快速**: 测试应该快速执行（通常在毫秒级别）
- **隔离**: 使用模拟对象隔离外部依赖
- **可重复**: 测试结果应该是一致的

### 4.2 UserService单元测试示例

```csharp
public class UserServiceTests
{
    private readonly Mock<IUserRepository> _mockUserRepository;
    private readonly Mock<IMapper> _mockMapper;
    private readonly Mock<ILogger<UserService>> _mockLogger;
    private readonly UserService _userService;

    public UserServiceTests()
    {
        _mockUserRepository = new Mock<IUserRepository>();
        _mockMapper = new Mock<IMapper>();
        _mockLogger = new Mock<ILogger<UserService>>();
        
        _userService = new UserService(
            _mockUserRepository.Object,
            _mockMapper.Object,
            _mockLogger.Object);
    }

    [Fact]
    public async Task GetUserByIdAsync_WhenUserExists_ReturnsUserDto()
    {
        // Arrange
        var userId = 1;
        var user = new User { Id = userId, Name = "John Doe", Email = "john@example.com" };
        var userDto = new UserDto { Id = userId, Name = "John Doe", Email = "john@example.com" };
        
        _mockUserRepository.Setup(repo => repo.GetByIdAsync(userId))
            .ReturnsAsync(user);
        _mockMapper.Setup(mapper => mapper.Map<UserDto>(user))
            .Returns(userDto);

        // Act
        var result = await _userService.GetUserByIdAsync(userId);

        // Assert
        result.Should().NotBeNull();
        result!.Id.Should().Be(userId);
        result.Name.Should().Be("John Doe");
        result.Email.Should().Be("john@example.com");
        
        _mockUserRepository.Verify(repo => repo.GetByIdAsync(userId), Times.Once);
        _mockMapper.Verify(mapper => mapper.Map<UserDto>(user), Times.Once);
    }

    [Fact]
    public async Task GetUserByIdAsync_WhenUserDoesNotExist_ReturnsNull()
    {
        // Arrange
        var userId = 999;
        
        _mockUserRepository.Setup(repo => repo.GetByIdAsync(userId))
            .ReturnsAsync((User?)null);

        // Act
        var result = await _userService.GetUserByIdAsync(userId);

        // Assert
        result.Should().BeNull();
        
        _mockUserRepository.Verify(repo => repo.GetByIdAsync(userId), Times.Once);
        _mockMapper.Verify(mapper => mapper.Map<UserDto>(It.IsAny<User>()), Times.Never);
    }

    [Fact]
    public async Task CreateUserAsync_WithValidData_CreatesUserSuccessfully()
    {
        // Arrange
        var createUserDto = new UserCreateDto 
        { 
            Name = "Jane Smith", 
            Email = "jane@example.com", 
            Password = "Password123!" 
        };
        
        var user = new User 
        { 
            Id = 1, 
            Name = "Jane Smith", 
            Email = "jane@example.com" 
        };
        
        var userDto = new UserDto 
        { 
            Id = 1, 
            Name = "Jane Smith", 
            Email = "jane@example.com" 
        };

        _mockUserRepository.Setup(repo => repo.GetByEmailAsync(It.IsAny<string>()))
            .ReturnsAsync((User?)null);
        _mockMapper.Setup(mapper => mapper.Map<User>(createUserDto))
            .Returns(user);
        _mockUserRepository.Setup(repo => repo.AddAsync(user))
            .ReturnsAsync(user);
        _mockUserRepository.Setup(repo => repo.SaveChangesAsync())
            .Returns(Task.CompletedTask);
        _mockMapper.Setup(mapper => mapper.Map<UserDto>(user))
            .Returns(userDto);

        // Act
        var result = await _userService.CreateUserAsync(createUserDto);

        // Assert
        result.Should().NotBeNull();
        result.Id.Should().Be(1);
        result.Name.Should().Be("Jane Smith");
        result.Email.Should().Be("jane@example.com");
        
        _mockUserRepository.Verify(repo => repo.GetByEmailAsync("jane@example.com"), Times.Once);
        _mockUserRepository.Verify(repo => repo.AddAsync(user), Times.Once);
        _mockUserRepository.Verify(repo => repo.SaveChangesAsync(), Times.Once);
    }

    [Fact]
    public async Task CreateUserAsync_WithExistingEmail_ThrowsArgumentException()
    {
        // Arrange
        var createUserDto = new UserCreateDto 
        { 
            Name = "Jane Smith", 
            Email = "existing@example.com", 
            Password = "Password123!" 
        };
        
        var existingUser = new User 
        { 
            Id = 1, 
            Name = "Existing User", 
            Email = "existing@example.com" 
        };

        _mockUserRepository.Setup(repo => repo.GetByEmailAsync("existing@example.com"))
            .ReturnsAsync(existingUser);

        // Act & Assert
        var exception = await Assert.ThrowsAsync<ArgumentException>(
            () => _userService.CreateUserAsync(createUserDto));
        
        exception.Message.Should().Contain("邮箱");
        exception.Message.Should().Contain("已被使用");
        
        _mockUserRepository.Verify(repo => repo.GetByEmailAsync("existing@example.com"), Times.Once);
        _mockUserRepository.Verify(repo => repo.AddAsync(It.IsAny<User>()), Times.Never);
    }
}
```

### 4.3 控制器单元测试示例

```csharp
public class UsersControllerTests
{
    private readonly Mock<IUserService> _mockUserService;
    private readonly Mock<ILogger<UsersController>> _mockLogger;
    private readonly UsersController _controller;

    public UsersControllerTests()
    {
        _mockUserService = new Mock<IUserService>();
        _mockLogger = new Mock<ILogger<UsersController>>();
        
        _controller = new UsersController(
            _mockUserService.Object, 
            _mockLogger.Object);
    }

    [Fact]
    public async Task GetUsersAsync_ValidRequest_ReturnsOkResult()
    {
        // Arrange
        var users = new List<UserDto>
        {
            new UserDto { Id = 1, Name = "John Doe", Email = "john@example.com" },
            new UserDto { Id = 2, Name = "Jane Smith", Email = "jane@example.com" }
        };
        
        var paginatedResponse = new PaginatedResponse<UserDto>
        {
            Items = users,
            Total = 2,
            Page = 1,
            PageSize = 10,
            TotalPages = 1
        };

        _mockUserService.Setup(service => service.GetUsersAsync(1, 10, null))
            .ReturnsAsync(paginatedResponse);

        // Act
        var result = await _controller.GetUsersAsync(1, 10, null);

        // Assert
        var okResult = result.Result as OkObjectResult;
        okResult.Should().NotBeNull();
        okResult!.StatusCode.Should().Be(200);
        
        var apiResponse = okResult.Value as ApiResponse<PaginatedResponse<UserDto>>;
        apiResponse.Should().NotBeNull();
        apiResponse!.Success.Should().BeTrue();
        apiResponse.Data!.Items.Count().Should().Be(2);
    }

    [Fact]
    public async Task GetUserByIdAsync_UserExists_ReturnsOkResult()
    {
        // Arrange
        var userId = 1;
        var userDto = new UserDto { Id = userId, Name = "John Doe", Email = "john@example.com" };

        _mockUserService.Setup(service => service.GetUserByIdAsync(userId))
            .ReturnsAsync(userDto);

        // Act
        var result = await _controller.GetUserByIdAsync(userId);

        // Assert
        var okResult = result.Result as OkObjectResult;
        okResult.Should().NotBeNull();
        okResult!.StatusCode.Should().Be(200);
        
        var apiResponse = okResult.Value as ApiResponse<UserDto>;
        apiResponse.Should().NotBeNull();
        apiResponse!.Success.Should().BeTrue();
        apiResponse.Data!.Id.Should().Be(userId);
    }

    [Fact]
    public async Task GetUserByIdAsync_UserDoesNotExist_ReturnsNotFound()
    {
        // Arrange
        var userId = 999;

        _mockUserService.Setup(service => service.GetUserByIdAsync(userId))
            .ReturnsAsync((UserDto?)null);

        // Act
        var result = await _controller.GetUserByIdAsync(userId);

        // Assert
        var notFoundResult = result.Result as NotFoundObjectResult;
        notFoundResult.Should().NotBeNull();
        notFoundResult!.StatusCode.Should().Be(404);
        
        var apiResponse = notFoundResult.Value as ApiResponse<UserDto>;
        apiResponse.Should().NotBeNull();
        apiResponse!.Success.Should().BeFalse();
    }
}
```

## 5. 集成测试

### 5.1 集成测试原则

- **真实环境**: 使用真实的数据库和其他依赖

### 5.2 WebApplicationFactory集成测试示例

```csharp
public class UsersIntegrationTests : IClassFixture<CustomWebApplicationFactory<Program>>
{
    private readonly CustomWebApplicationFactory<Program> _factory;
    private readonly HttpClient _client;

    public UsersIntegrationTests(CustomWebApplicationFactory<Program> factory)
    {
        _factory = factory;
        _client = factory.CreateClient();
    }

    [Fact]
    public async Task GetUsers_EndpointReturnsSuccessAndCorrectContentType()
    {
        // Arrange & Act
        var response = await _client.GetAsync("/api/users");

        // Assert
        response.EnsureSuccessStatusCode(); // Status Code 200-299
        response.Content.Headers.ContentType!.ToString()
            .Should().Contain("application/json");
    }

    [Fact]
    public async Task GetUserById_WithValidId_ReturnsUser()
    {
        // Arrange
        var userId = 1;
        
        // Act
        var response = await _client.GetAsync($"/api/users/{userId}");

        // Assert
        response.EnsureSuccessStatusCode();
        var responseString = await response.Content.ReadAsStringAsync();
        var apiResponse = JsonSerializer.Deserialize<ApiResponse<UserDto>>(
            responseString, 
            new JsonSerializerOptions { PropertyNameCaseInsensitive = true });
        
        apiResponse.Should().NotBeNull();
        apiResponse!.Success.Should().BeTrue();
        apiResponse.Data!.Id.Should().Be(userId);
    }

    [Fact]
    public async Task CreateUser_WithValidData_CreatesUser()
    {
        // Arrange
        var newUser = new
        {
            Name = "Integration Test User",
            Email = "integration@test.com",
            Password = "Password123!"
        };
        
        var json = JsonSerializer.Serialize(newUser);
        var content = new StringContent(json, Encoding.UTF8, "application/json");

        // Act
        var response = await _client.PostAsync("/api/users", content);

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.Created);
        
        var responseString = await response.Content.ReadAsStringAsync();
        var apiResponse = JsonSerializer.Deserialize<ApiResponse<UserDto>>(
            responseString,
            new JsonSerializerOptions { PropertyNameCaseInsensitive = true });
        
        apiResponse.Should().NotBeNull();
        apiResponse!.Success.Should().BeTrue();
        apiResponse.Data!.Name.Should().Be("Integration Test User");
        apiResponse.Data.Email.Should().Be("integration@test.com");
    }
}

public class CustomWebApplicationFactory<TProgram> : WebApplicationFactory<TProgram> 
    where TProgram : class
{
    protected override void ConfigureWebHost(IWebHostBuilder builder)
    {
        builder.ConfigureServices(services =>
        {
            // 移除原有的数据库上下文配置
            var descriptor = services.SingleOrDefault(
                d => d.ServiceType == typeof(DbContextOptions<AppDbContext>));
            
            if (descriptor != null)
            {
                services.Remove(descriptor);
            }

            // 添加内存数据库用于测试
            services.AddDbContext<AppDbContext>(options =>
            {
                options.UseInMemoryDatabase("TestDb");
            });

            // 构建服务提供者并初始化数据库
            var serviceProvider = services.BuildServiceProvider();

            using var scope = serviceProvider.CreateScope();
            var context = scope.ServiceProvider.GetRequiredService<AppDbContext>();
            
            // 确保数据库被创建
            context.Database.EnsureCreated();
            
            // 添加测试数据
            SeedTestData(context);
        });
    }

    private void SeedTestData(AppDbContext context)
    {
        if (!context.Users.Any())
        {
            var testUser = new User
            {
                Name = "Test User",
                Email = "test@example.com",
                PasswordHash = BCrypt.Net.BCrypt.HashPassword("Password123!"),
                IsActive = true,
                CreatedAt = DateTime.UtcNow
            };
            
            context.Users.Add(testUser);
            context.SaveChanges();
        }
    }
}
```

## 7. 测试最佳实践

### 7.1 AAA模式 (Arrange, Act, Assert)

```csharp
[Fact]
public async Task MethodUnderTest_Scenario_ExpectedBehavior()
{
    // Arrange - 准备测试数据和依赖
    var input = new InputModel { Value = "test" };
    var expected = "expected result";
    
    // Act - 执行被测方法
    var result = await _sut.ProcessAsync(input);
    
    // Assert - 验证结果
    result.Should().Be(expected);
}
```

public static class AssertionExtensions
{

        userDto.Name.Should().NotBeNullOrWhiteSpace();
